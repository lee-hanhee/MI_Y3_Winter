\subsection{Robustness Margins}
\begin{motivation}
    Approximate the GM and PM from the Bode plot: 
    \begin{itemize}
        \item $L(s)$ is a strictly proper rational fn. 
        \item $L(s)$ has no poles in $\mathbb{C}^+$ (no open loop variable poles)
    \end{itemize}
\end{motivation}

\end{definition}
